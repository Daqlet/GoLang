#define ll long long

bool good(map<int, int> &m) {
    bool gd = 1;
    for(auto i = m.begin(); i != m.end(); ++i) {
        auto j = i; ++j;
        while(j != m.end()) {
            if(!m[abs(i->first - j->first)]) {
                m[abs(i->first - j->first)] = 1;
                gd = 0;
            }
            ++j;
        }
    }
    return gd;
}
