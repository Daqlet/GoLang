#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    int n; cin >> n;
    ll sum = 0;
    priority_queue<int, vector<int>, greater<int>> pq;
    for(int i = 0, a; i < n; ++i) {
        cin >> a;
        sum += a; pq.push(a);
        while(sum < 0) {
            sum -= pq.top();
            pq.pop();
        }
    }
    cout << pq.size() << endl;
    return 0;
}
