#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    int t = 1; cin >> t;
    while(t--) {
        int a, b, c, d; cin >> a >> b >> c >> d;
        if(min(a, b) > max(c, d) || max(a, b) < min(c, d)) cout << "No" << endl;
        else cout << "Yes" << endl;
    }
    return 0;
}
